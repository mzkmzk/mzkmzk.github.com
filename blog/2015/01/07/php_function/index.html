php json数据返回多了一个空行

很大的一种概率是php中有空的字符返回了.懒得找的方式就是这样得到json后去掉一下前后空行

String.prototype.trim = function() {
var str = this,
str = str.replace(/^\s\s*/, ''),
ws = /\s/,
i = str.length;
while (ws.test(str.charAt(--i)));
return str.slice(0, i + 1);
}
